<script>
/**
 * 直播网站的首页，
 * 包含导航栏、直播列表、录播列表等信息
 *
 * @modules HomePage
 * @class HomePage
 */
import LivePicture from './LivePicture'
import RecordPicture from './RecordPicture'
import HomePageHeader from './HomePageHeader'
import PageFooter from './PageFooter'

export default {
    name: 'home-page',
    components: {
        LivePicture,
        RecordPicture,
        HomePageHeader,
        PageFooter
    },
    data: function () {
        return {
            rooms: [],
            option: 1
        }
    },
    /**
     * created函数，调用getRooms函数，
     * 获取房间信息
     *
     * @method created
     */
    created: function () {
        this.getRooms()
    },
    methods: {
        /**
         * 获取正在直播及录播房间信息
         *
         * @method getRooms
         */
        getRooms: function () {
            fetch('/getRooms/', {
                method: 'post',
                mode: 'cors',
                headers: {
                    'Content-Type': 'application/json, text/plain, */*',
                    'Accept': 'application/json'
                },
                body: JSON.stringify({ 'type': 1 })
            }).then((response) => response.json()).then((obj) => {
                this.rooms = obj.rooms
            })
        }
    }
}
</script>
